# Структура проекта

```
├── src/                                  # Исходники
│   ├── img/                              # Изображения
│   │   └── svg/                          # svg
│   ├── js/                               # Скрипты
│   │   └── main.js                       # Главный скрипт
│   ├── libs/                             # Плагины
│   ├── root/                             # Файлы, которые будут копироваться в корень проекта (faviocn.ico) и.т.д
│   ├── styles/                           # Стили проекта
│   │   ├── _fonts.scss               	  # Объявление шрифтов
│   │   ├── _libs.scss                	  # Стили библиотек
│   │   ├── _ui.scss                  	  # Стили для отдельных блоков
│   │   ├── _media.scss               	  # Стили для адаптива
│   │   └── main.scss                 	  # Основные стили
│   ├── templates/                        # Html блоки кода
│   │   ├── footer.html               	  
│   │   └── header.html                	  
│   ├── index.html               	  
│   └── ui.html      
```
